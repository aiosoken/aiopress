### 2.3. データフロー

#### 2.3.1. 資産アップロード・分析フロー

```
1. ユーザーがファイルをアップロード
   ↓
2. Next.js → Firebase Storage (ファイル保存)
   ↓
3. Storage トリガー → Cloud Function (onFileUpload)
   ↓
4. Cloud Function → Vision AI (OCR, ラベル検出)
   ↓
5. Cloud Function → Gemini API (詳細分析)
   ↓
6. Cloud Function → Firestore (分析結果保存)
   ↓
7. Cloud Function → Firestore (デザインシステム更新)
   ↓
8. Firestore → Next.js (リアルタイム更新)
```

#### 2.3.2. クリエイティブ生成フロー

```
1. ユーザーが生成指示を入力
   ↓
2. Next.js → Cloud Function (generateCreative)
   ↓
3. Cloud Function → Firestore (デザインシステム取得)
   ↓
4. Cloud Function → Gemini API / Imagen API (コンテンツ生成)
   ↓
5. Cloud Function → Firebase Storage (画像保存)
   ↓
6. Cloud Function → Firestore (生成物保存)
   ↓
7. Firestore → Next.js (結果表示)
```

## 7. セキュリティ設計

### 7.1. Firestoreセキュリティルール

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ヘルパー関数
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isBrandMember(brandId) {
      return isAuthenticated() && 
             exists(/databases/$(database)/documents/brandMembers/$(brandId + '_' + request.auth.uid));
    }
    
    function isBrandOwner(brandId) {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/brandMembers/$(brandId + '_' + request.auth.uid)).data.role == 'OWNER';
    }
    
    function isBrandAdmin(brandId) {
      let role = get(/databases/$(database)/documents/brandMembers/$(brandId + '_' + request.auth.uid)).data.role;
      return role == 'OWNER' || role == 'ADMIN';
    }
    
    // ユーザープロファイル
    match /users/{userId} {
      allow read, write: if isAuthenticated() && request.auth.uid == userId;
    }
    
    // ブランド
    match /brands/{brandId} {
      allow read: if isBrandMember(brandId);
      allow create: if isAuthenticated();
      allow update, delete: if isBrandOwner(brandId);
    }
    
    // ブランドメンバー
    match /brandMembers/{memberId} {
      allow read: if isAuthenticated();
      allow write: if isBrandOwner(resource.data.brandId);
    }
    
    // 資産
    match /assets/{assetId} {
      allow read: if isBrandMember(resource.data.brandId);
      allow create: if isAuthenticated() && isBrandMember(request.resource.data.brandId);
      allow update, delete: if isBrandMember(resource.data.brandId);
    }
    
    // デザインシステム
    match /designSystems/{brandId} {
      allow read: if isBrandMember(brandId);
      allow write: if isBrandOwner(brandId) || isBrandAdmin(brandId);
    }
    
    // クリエイティブ
    match /creatives/{creativeId} {
      allow read: if isBrandMember(resource.data.brandId);
      allow create: if isAuthenticated() && isBrandMember(request.resource.data.brandId);
      allow update, delete: if isBrandMember(resource.data.brandId);
    }
  }
}
```

### 7.3. Cloud Functions認証

すべてのHTTP Callable Functionsは認証を必須とする。

```typescript
export const generateCreative = functions
  .region('asia-northeast1')
  .https
  .onCall(async (data, context) => {
    // 認証チェック
    if (!context.auth) {
      throw new functions.https.HttpsError(
        'unauthenticated',
        '認証が必要です'
      );
    }
    
    // 権限チェック
    const isMember = await checkBrandMembership(
      context.auth.uid,
      data.brandId
    );
    
    if (!isMember) {
```

```typescript
        code: authCode,
        client_id: this.clientId,
```
